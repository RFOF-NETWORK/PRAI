        const wrapLabel = (label) => {
            const maxLen = 16;
            if (label.length <= maxLen) return label;
            let parts = [];
            for (let i = 0; i < label.length; i += maxLen) {
                parts.push(label.substring(i, i + maxLen));
            }
            return parts;
